grid.search <- function(gs_y.v, gs_lossf, gs_f, gs_true){
  par(mar = c(5, 5, 2, 2))
  
  # Initial bounds
  a <- min(gs_y.v) ; b <- max(gs_y.v) ; K <- 10 ; eps <- 1e-2
  iter.max <- 1e2
  
  plot(Vectorize(function(t) gs_f(gs_y.v, t, gs_lossf)), a, b, ylab = expression(f(mu)),
       xlab = expression(mu))
  
  # Step.2 ~ 5
  trace.v <- rep(NA, iter.max)
  for (j in 1:iter.max){
    trace.v[j] <- gs_f(y.v, a, gs_lossf)
    x.v <- seq(a, b, length.out = K)
    f.v <- rep(NA, K)
    for (i in 1:K) {
    f.v[i] <- gs_f(y.v, x.v[i], gs_lossf)
    }
    opt <- which.min(f.v) # i*
    a <- x.v[opt - 1]
    b <- x.v[opt + 1]
    if (b - a < eps) break
  }
  points(c(a, b), c(f(gs_y.v, a, gs_lossf), f(gs_y.v, b, gs_lossf)), pch = 19, cex = .5)
  legend('topleft', '(a,f(a))', pch = 19)
  plot(log(trace.v[1:j],exp(10)), ylab = 'log trace') # Is trace decreasing in iteration?
  return(list(a = a, b = b, mustar = gs_true))
}
