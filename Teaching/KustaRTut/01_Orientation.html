Coding Test

set.seed(0536)
org <- split(sample(1:30, 18, replace = F), 1:3)
a <- org$`1`
b <- org$`2`
c <- org$`3`
int <- sort(c(a, b, c))
eg <- combn(int, 2)
trues <- matrix(0, nrow = 2)
for (i in 1:ncol(eg)) {
  interval.v <- eg[1, i]:eg[2, i]
  test <- all(any(a %in% interval.v),any(b %in% interval.v),any(c %in% interval.v))
  if (test == 1) {trues <- cbind(trues, eg[,i])}
}
trues <- trues[,-1]
trues[,which.min(trues[2,] - trues[1,])]
## [1] 4 6
